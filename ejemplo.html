Ejemplo

Código del semáforo

// Declaración de las variables que representan los pines donde están conectados los LEDs
const int ledRojo = 13;      // LED rojo conectado al pin digital 13
const int ledAmarillo = 12;  // LED amarillo conectado al pin digital 12
const int ledVerde = 11;     // LED verde conectado al pin digital 11


// Función de configuración que se ejecuta una sola vez al iniciar el programa
void setup() {
  // Se configuran los pines como salidas para poder encender y apagar los LEDs
  pinMode(ledRojo, OUTPUT);
  pinMode(ledAmarillo, OUTPUT);
  pinMode(ledVerde, OUTPUT);
}


// Función principal que se ejecuta en bucle (repetidamente)
void loop() {
  // Encender el LED verde (simula luz verde del semáforo)
  digitalWrite(ledVerde, HIGH);    
  delay(5000);                     // Mantener encendido durante 5 segundos
  digitalWrite(ledVerde, LOW);     // Apagar el LED verde


  // Encender el LED amarillo (simula luz amarilla del semáforo)
  digitalWrite(ledAmarillo, HIGH); 
  delay(2000);                     // Mantener encendido durante 2 segundos
  digitalWrite(ledAmarillo, LOW);  // Apagar el LED amarillo

  // Encender el LED rojo (simula luz roja del semáforo)
  digitalWrite(ledRojo, HIGH);     
  delay(5000);                     // Mantener encendido durante 5 segundos
  digitalWrite(ledRojo, LOW);      // Apagar el LED rojo


  // Luego, el ciclo se repite indefinidamente (verde → amarillo → rojo → repetir)
}
